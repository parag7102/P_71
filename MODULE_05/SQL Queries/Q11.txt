    CNo int NOT NULL PRIMARY KEY,
    CName nvarchar(50) NOT NULL,
    City nvarchar(50) NOT NULL,
    Rating int NOT NULL,

    FOREIGN KEY (SNo) REFERENCES Salesperson(SNo)
